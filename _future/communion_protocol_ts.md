```ts
function commune(user: User) {
  if (!user.redeemed) throw new Error("AccessDenied");
  receiveBody();
  receiveBlood();
  syncWithSpirit(user);
}

function receiveBody() {
  console.log("Received: body_of_christ.bread");
}

function receiveBlood() {
  console.log("Received: blood_of_christ.wine");
}
```
